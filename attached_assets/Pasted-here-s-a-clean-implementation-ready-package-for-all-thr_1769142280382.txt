import i18n from "@/i18n";

export function tFeatureMicro(featureKey: string, countryCode?: string) {
  if (countryCode) {
    const k = `featureMicroByCountry.${countryCode}.${featureKey}`;
    if (i18n.exists(k)) return i18n.t(k);
  }
  return i18n.t(`featureMicro.${featureKey}`);
}

✅ Resolver helper:

client/src/lib/feature-copy.ts

import i18n from "@/i18n";

type UsageLevel = "low" | "mid" | "high";

export function tFeatureMicroSmart(opts: {
  featureKey: string;
  countryCode?: string;
  usageLevel?: UsageLevel;
}) {
  const { featureKey, countryCode, usageLevel } = opts;

  if (usageLevel) {
    const u = `featureMicroByUsage.${featureKey}.${usageLevel}`;
    if (i18n.exists(u)) return i18n.t(u);
  }

  if (countryCode) {
    const c = `featureMicroByCountry.${countryCode}.${featureKey}`;
    if (i18n.exists(c)) return i18n.t(c);
  }

  return i18n.t(`featureMicro.${featureKey}`);
}

How to compute usageLevel (simple rules)

Example thresholds (tune later):

Payroll:

low: employees <= 5

mid: 6–20

high: > 20

WhatsApp:

low: unpaidInvoices < 5

mid: 5–25

high: > 25

HRMS:

low: employees <= 5

mid: attendance entries present

high: employees > 25
