Implementation tip (to avoid duplicated logic)

Create one helper:

export function getEmployeeDirectoryAbilities({ entitlements, permissionService, userId, tenantId }) {
  const hasModule = entitlements.hasAnyAddon(["hrms", "payroll"]);

  return {
    canView: hasModule && permissionService.hasPermission(userId, tenantId, "staff:read"),
    canCreate: hasModule && permissionService.hasPermission(userId, tenantId, "staff:create"),
    canEdit: hasModule && permissionService.hasPermission(userId, tenantId, "staff:update"),
    canDeactivate: hasModule && permissionService.hasPermission(userId, tenantId, "staff:update"),
    canDelete: hasModule && permissionService.hasPermission(userId, tenantId, "staff:delete"),
  };
}
