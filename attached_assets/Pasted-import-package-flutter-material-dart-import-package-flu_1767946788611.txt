import 'package:flutter/material.dart';
import 'package:flutter_bloc/flutter_bloc.dart';
import '../blocs/auth_bloc.dart';
import '../repositories/auth_repository.dart';

class SignupPage extends StatefulWidget {
  const SignupPage({Key? key}) : super(key: key);

  @override
  State<SignupPage> createState() => _SignupPageState();
}

class _SignupPageState extends State<SignupPage> {
  final _formKey = GlobalKey<FormState>();
  String tenantName = '';
  String tenantSlug = '';
  String businessType = '';
  String adminEmail = '';
  String adminPassword = '';

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text("Tenant Signup")),
      body: Form(
        key: _formKey,
        child: ListView(
          padding: const EdgeInsets.all(16),
          children: [
            TextFormField(
              decoration: const InputDecoration(labelText: "Tenant Name"),
              onChanged: (v) => tenantName = v,
              validator: (v) => v!.isEmpty ? "Required" : null,
            ),
            TextFormField(
              decoration: const InputDecoration(labelText: "Tenant Slug"),
              onChanged: (v) => tenantSlug = v,
              validator: (v) => v!.isEmpty ? "Required" : null,
            ),
            TextFormField(
              decoration: const InputDecoration(labelText: "Business Type"),
              onChanged: (v) => businessType = v,
              validator: (v) => v!.isEmpty ? "Required" : null,
            ),
            TextFormField(
              decoration: const InputDecoration(labelText: "Admin Email"),
              onChanged: (v) => adminEmail = v,
              validator: (v) => !v!.contains('@') ? "Invalid Email" : null,
            ),
            TextFormField(
              decoration: const InputDecoration(labelText: "Password"),
              obscureText: true,
              onChanged: (v) => adminPassword = v,
              validator: (v) => v!.length < 6 ? "Min 6 chars" : null,
            ),
            const SizedBox(height: 16),
            ElevatedButton(
              onPressed: () {
                if (_formKey.currentState!.validate()) {
                  context.read<AuthBloc>().add(SignupRequested(
                    tenantName: tenantName,
                    tenantSlug: tenantSlug,
                    businessType: businessType,
                    adminEmail: adminEmail,
                    adminPassword: adminPassword
                  ));
                }
              },
              child: const Text("Sign Up"),
            )
          ],
        ),
      ),
    );
  }
}
